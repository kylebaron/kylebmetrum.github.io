library(mrgsolve)
code <- '
$PARAM KAMAX = 1.1, CL=1, VC=20, KAT50 = 2

$CMT GUT CENT

$GLOBAL
double TAD = 0;
double KA = 0;
double OFFSET = 0;


$MAIN if(EVID==1) OFFSET = TIME;

$ODE 

double KAT  = SOLVERTIME - OFFSET;
KA = KAMAX*KAT/(KAT50 + KAT);
dxdt_GUT = -KA*GUT;
dxdt_CENT = KA*GUT - (CL/VC)*CENT;

$CAPTURE KA
'
mod <- mread(code=code, model="timeka")

## done.
